const express = require('express');
const multer = require('multer');

const cors = require('cors');

// Sicherer Upload-Ordner

// Middleware
app.use(cors());
app.use(express.static('public'));
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// Multer f√ºr Datei-Uploads konfigurieren
const storage = multer.diskStorage({
  destination: (req, file, cb) => cb(null, UPLOAD_DIR),
  filename: (req, file, cb) => cb(null, `upload_${Date.now()}_${path.extname(file.originalname)}`)
});

const upload = multer({ 
  storage: storage, 
  limits: { fileSize: 100 * 1024 * 1024 }, // 100MB Limit
  fileFilter: (req, file, cb) => {
    if (path.extname(file.originalname).toLowerCase() === '.zip') cb(null, true);
    else cb(new Error('Nur ZIP-Dateien erlaubt'));
  }
});

// Routes
app.get('/', (req, res) => res.sendFile(path.join(__dirname, 'public', 'index.html')));

app.get('/api/health', (req, res) => res.json({ status: 'OK', service: 'ZIP-SHIP API' }));

// Haupt-Deploy-Route
app.post('/api/deploy', upload.single('zipfile'), async (req, res) => {
  console.log('Deploy-Anfrage erhalten:', { file: req.file?.originalname, repoName: req.body.repoName });
  
  try {
    // 1. Validierung
    if (!req.file) throw new Error('Keine ZIP-Datei hochgeladen');
    if (!req.body.repoName || req.body.repoName.trim() === '') throw new Error('Repository-Name fehlt');
    if (!req.body.githubToken && !process.env.GITHUB_TOKEN) throw new Error('GitHub Token nicht gefunden. Bitte in .env setzen oder im Formular angeben.');

    const repoName = req.body.repoName.trim();
    const githubToken = req.body.githubToken || process.env.GITHUB_TOKEN;
    const makePrivate = req.body.makePrivate === 'true';

    // 2. ZIP-Verarbeitung
    const extractedFiles = await processZip(req.file.path);
    console.log(`‚úÖ ZIP verarbeitet: ${extractedFiles.length} Dateien extrahiert`);

    // 3. GitHub-Operation
    const repoUrl = await createRepoAndPush(repoName, extractedFiles, githubToken, makePrivate);
    
    // 4. Tempor√§re Dateien bereinigen
    fs.unlink(req.file.path, (err) => err && console.error('Fehler beim L√∂schen:', err));

    // 5. Erfolgsantwort
    res.json({
      success: true,
      message: `Repository "${repoName}" erfolgreich erstellt und gepusht!`,
      repoUrl: repoUrl,
      fileCount: extractedFiles.length,
      timestamp: new Date().toISOString()
    });

  } catch (error) {
    console.error('‚ùå Fehler im Deploy-Prozess:', error.message);
    
    // Hochgeladene Datei bei Fehler l√∂schen
    if (req.file && fs.existsSync(req.file.path)) fs.unlink(req.file.path, () => {});

    res.status(500).json({ 
      success: false, 
      error: error.message,
      details: process.env.NODE_ENV === 'development' ? error.stack : undefined 
    });
  }

// utils/zipHandler.js - Sicherer ZIP-Handler mit Zip-Slip-Schutz
const AdmZip = require('adm-zip');
const path = require('path');
const fs = require('fs').promises;

async function processZip(zipFilePath) {
  console.log(`Verarbeite ZIP: ${zipFilePath}`);
  const zip = new AdmZip(zipFilePath);
  const extractDir = path.join(__dirname, '..', 'tmp_extract_' + Date.now());
  const extractedFiles = [];

  try {
    // Extraktionsordner erstellen
    await fs.mkdir(extractDir, { recursive: true });

    // ZIP-Eintr√§ge durchgehen
    const zipEntries = zip.getEntries();
    console.log(`Gefunden: ${zipEntries.length} Eintr√§ge im ZIP`);

    for (const entry of zipEntries) {
      // 1. KRITISCH: Zip-Slip-Angriff verhindern
      const entryName = entry.entryName.replace(/\\/g, '/'); // Windows-Pfade normalisieren
      const targetPath = path.resolve(path.join(extractDir, entryName));
      const normalizedExtractDir = path.resolve(extractDir) + path.sep;

      if (!targetPath.startsWith(normalizedExtractDir)) {
        throw new Error(`Sicherheitsverletzung: Unzul√§ssiger Pfad "${entryName}" versucht, au√üerhalb des Extraktionsordners zu schreiben.`);
      }

      // 2. Unerw√ºnschte Dateien/Ordner filtern
      const ignoredPatterns = [
        /node_modules\//, /\.git\//, /__pycache__\//,
        /\.DS_Store$/, /Thumbs\.db$/, /desktop\.ini$/,
        /\.env$/, /\.npmrc$/, /\.yarnrc$/
      ];

      if (ignoredPatterns.some(pattern => pattern.test(entryName))) {
        console.log(`‚ö†Ô∏è  Ignoriere: ${entryName}`);
        continue;
      }

      // 3. Datei verarbeiten (nur Dateien, keine Ordner)
      if (!entry.isDirectory) {
        try {
          let content;
          // Versuche, Textdateien zu lesen, andere als Base64
          if (/\.(txt|js|json|html|css|py|java|c|cpp|md|yml|yaml|xml)$/i.test(entryName)) {
            content = zip.readAsText(entry, 'utf8');
          } else {
            content = zip.readFile(entry).toString('base64');
          }

          extractedFiles.push({
            path: entryName,
            content: content,
            isBinary: !/\.(txt|js|json|html|css|py|java|c|cpp|md|yml|yaml|xml)$/i.test(entryName)
          });
        } catch (readErr) {
          console.warn(`Konnte Datei ${entryName} nicht lesen:`, readErr.message);
        }
      }
    }

    console.log(`‚úÖ Erfolgreich extrahiert: ${extractedFiles.length} Dateien`);
    return extractedFiles;

  } catch (error) {
    console.error('Fehler in processZip:', error);
    throw error;
  } finally {
    // Aufr√§umen: Extraktionsordner l√∂schen
    try {
      await fs.rm(extractDir, { recursive: true, force: true });
    } catch (cleanupErr) {
      console.warn('Warnung: Extraktionsordner konnte nicht gel√∂scht werden:', cleanupErr.message);
    }
  }
}

module.exports = { processZip };

// utils/githubService.js - GitHub API Service
const { Octokit } = require('octokit');

async function createRepoAndPush(repoName, files, githubToken, isPrivate = false) {
  console.log(`Erstelle GitHub Repo: ${repoName} (${isPrivate ? 'private' : 'public'})`);
  
  try {
    // Octokit mit Token initialisieren
    const octokit = new Octokit({ auth: githubToken });

    // 1. Repository erstellen
    const createRepoResponse = await octokit.request('POST /user/repos', {
      name: repoName,
      description: 'Deployed with ‚ù§Ô∏è via ZIP-SHIP',
      private: isPrivate,
      auto_init: false,
      has_issues: false,
      has_wiki: false,
      has_downloads: false
    });

    const repoData = createRepoResponse.data;
    const owner = repoData.owner.login;
    const repoFullName = repoData.full_name;
    console.log(`‚úÖ Repository erstellt: ${repoFullName}`);

    // 2. F√ºr jede Datei einen Git-Blob erstellen
    console.log(`Erstelle Blobs f√ºr ${files.length} Dateien...`);
    const treeItems = [];

    for (const file of files) {
      try {
        const blobData = {
          owner,
          repo: repoName,
          content: file.isBinary ? file.content : Buffer.from(file.content).toString('base64'),
          encoding: file.isBinary ? 'base64' : 'utf-8'
        };

        const blobResponse = await octokit.request('POST /repos/{owner}/{repo}/git/blobs', blobData);
        
        treeItems.push({
          path: file.path,
          mode: '100644', // Normale Datei
          type: 'blob',
          sha: blobResponse.data.sha
        });
      } catch (blobErr) {
        console.warn(`Blob f√ºr ${file.path} fehlgeschlagen:`, blobErr.message);
      }
    }

    // 3. Git-Tree erstellen
    console.log('Erstelle Git Tree...');
    const treeResponse = await octokit.request('POST /repos/{owner}/{repo}/git/trees', {
      owner,
      repo: repoName,
      tree: treeItems,
      base_tree: null
    });
    const treeSha = treeResponse.data.sha;

    // 4. Commit erstellen
    console.log('Erstelle Commit...');
    const commitResponse = await octokit.request('POST /repos/{owner}/{repo}/git/commits', {
      owner,
      repo: repoName,
      message: 'üöÄ Initial commit deployed by ZIP-SHIP\n\nAutomatically deployed from ZIP file.',
      tree: treeSha,
      parents: []
    });
    const commitSha = commitResponse.data.sha;

    // 5. Referenz (Branch) aktualisieren
    console.log('Aktualisiere Branch...');
    await octokit.request('PATCH /repos/{owner}/{repo}/git/refs/heads/main', {
      owner,
      repo: repoName,
      sha: commitSha
    });

    console.log(`‚úÖ Erfolgreich gepusht: ${treeItems.length} Dateien`);
    return repoData.html_url;

  } catch (error) {
    console.error('GitHub API Fehler:', error.message);
    
    // Benutzerfreundliche Fehlermeldungen
    if (error.status === 401) throw new Error('Ung√ºltiger GitHub Token. Bitte √ºberpr√ºfen Sie Ihren Personal Access Token.');
    if (error.status === 422) throw new Error(`Repository "${repoName}" existiert bereits oder Name ist ung√ºltig.`);
    if (error.status === 403) throw new Error('API-Ratelimit erreicht oder keine Berechtigung.');
    
    throw new Error(`GitHub Fehler: ${error.message}`);
  }
}

module.exports = { createRepoAndPush };

    function formatBytes(bytes, decimals = 2) {
        if (bytes === 0) return '0 Bytes';
        const k = 1024;
        const dm = decimals < 0 ? 0 : decimals;
        const sizes = ['Bytes', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i];
    }
